整个程序想实现两线程间通过消息队列进行通信，分为两个部分：
1.使用rt_mq_create()创建消息队列。
2.分别创建动态线程及其入口函数。

```cpp

rt_mq_t mq = RT_NULL;
uint32_t send_data1 = 1;
uint32_t send_data2 = 2;

static rt_thread_t Rx_thread = RT_NULL;
static void Rx_thread_entry(void* parameter)
{	
    rt_err_t uwRet = RT_EOK;
    uint32_t r_queue;
    while(1)
    {
      uwRet = rt_mq_recv(mq,
                         &r_queue,
                         sizeof(r_queue),
                         RT_WAITING_FOREVER);
      if (RT_EOK ==uwRet)
      {
        rt_kprintf("本次接收到的数据是：%d\n", r_queue);
        if (r_queue == 1)
        {
          LED1_TOGGLE;
        }
        else
        {
          LED2_TOGGLE;
        }
        
      }
      else
      {
        rt_kprintf("数据接收出错，错误代码： 0x%1x\n", uwRet);
      }
      rt_thread_delay(200);
    }
}

static rt_thread_t Tx_thread = RT_NULL;
static void Tx_thread_entry(void* parameter)
{
  rt_err_t uwRet = RT_EOK;

  while (1)
  {
    if (Key_Scan(KEY1_GPIO_PORT, KEY1_PIN) == KEY_ON)
    {
      uwRet = rt_mq_send(mq,
                        &send_data1,
                        sizeof(send_data1));

      if (uwRet != RT_EOK)
      {
        rt_kprintf("数据不能发送到消息队列：错误代码：%lx\n");
      }
    }
    if (Key_Scan(KEY2_GPIO_PORT, KEY2_PIN) == KEY_ON)
    {
      uwRet = rt_mq_send(mq,
                        &send_data2,
                        sizeof(send_data2));
      if (uwRet != RT_EOK)
      {
        rt_kprintf("数据不能发送到消息队列：错误代码：%lx\n");
      }
    }
    rt_thread_delay(20);
  }

}

int main(void)
{	
  mq = rt_mq_create("mq",
                     40,
                     20,
                     RT_IPC_FLAG_FIFO);
  if (mq != RT_NULL)
      rt_kprintf("massage queue create successfully!\n");\
  else
  {
    return -1;
  }
                
	Rx_thread =                          
    rt_thread_create( "Rx",              
                      Rx_thread_entry,   
                      RT_NULL,             
                      512,                
                      3,                   
                      20);                
                   

   if (Rx_thread != RT_NULL)
        rt_thread_startup(Rx_thread);
    else
        return -1;


  Tx_thread = 
    rt_thread_create("Tx",
                      Tx_thread_entry,
                      RT_NULL,
                      512,
                      2,
                      20);
  
  if (Tx_thread!=RT_NULL)
        rt_thread_startup(Tx_thread);
  else 
        return -1;
}

```


## 要注意的点：
1.rtt在使用ipc前要在rtconfig.h文件中开启对应选项。
2.(void*)类型的函数参数应当传入变量的地址。
